class Teste4{
int idade;
string nome, sobrenome, texto;
float bonus, salario, salarioLiquido;

{
repeat{
write("Digite o seu nome: ");
read(nome);
write("Digite o seu sobrenome");
read(sobrenome);
write("Digite a sua idade: ");
read(idade);
write("Digite o salario: ");
read(salario);
write("Digite o bonus: ");
read(bonus);

if (idade > 0 && salario > 0 and bonus >= 0){
salarioLiquido := (salario - salario * 0.11) + (bonus - 109.0);
texto := nome + " " + sobrenome + ":";
write(texto);
write("Salario liquido: ");
write(salarioLiquido);
};
} until (idade <= 0 or salario <= 0 or bonus < 0);
}
}
